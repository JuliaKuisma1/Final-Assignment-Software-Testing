  6 const readline = require('readline');
  7 const chalk = require('chalk');
  8 const figlet = require('figlet');
  9 let colorArray = [ "black", "red", "green", "yellow", "blue", "gray", "magenta", "cyan", "white" ];
 10 
 11 /** @constructor for Question object. */
 12 function Question() { };
 13 question = new Question();
 14 
 15 /** 
 16  * promptUser function displays first user question. 
 17  */
 18 Question.prototype.promptUser = function promptUser() {
 19     if (typeof rl == "undefined") {
 20         const rl = question.createReadlineInterface();
 21     }
 22     rl.question('Enter your input option (a, b, c): ', answer => {
 23         if (answer === 'a') {
 24             question.displayArray(colorArray);
 25             question.colorQuestion();
 26         }
 27         else if (answer === 'b') {
 28             question.sportQuestion();
 29         }
 30         else if (answer === 'c') {
 31             question.shutDown();
 32         }
 33         else {
 34             question.displayMessage('Please enter input option a, b or c, instead of',answer, 'error');
 35         }
 36     });
 37 };
 38 
 39 /** 
 40  * createReadlineInterface creates object rl that is used on user inputs.  
 41  */
 42 Question.prototype.createReadlineInterface = () => {
 43     return rl = readline.createInterface({
 44         input: process.stdin,
 45         output: process.stdout
 46     });
 47 }
 48 
 49 /** 
 50  * colorQuestion displays color options to user and waits for answer.
 51  * Then it calls function validate to check if color is valid.
 52  */
 53 Question.prototype.colorQuestion = () => {
 54     rl.question('Pick a color: ', answer => {
 55         question.validate(answer, colorArray);
 56     });
 57 };
 58 
 59 /** 
 60  * sportQuestion displays question about sports and waits for user input.
 61  * Then it calls function displayMessage to display answer and asks first question again. 
 62  */
 63 Question.prototype.sportQuestion = () => {
 64     rl.question('Enter your favorite sport: ', (answer) => {
 65         question.displayMessage('Your favorite sport:', answer);
 66         question.promptUser();
 67     });
 68 }
 69 
 70 /** 
 71  * shutDown function displays user message and closes readline interface.
 72  */
 73 Question.prototype.shutDown = () => {
 74     console.log(chalk.gray.bold('Shutting down...\n')+chalk.gray(figlet.textSync('ZzZzZz...')));
 75     rl.close();
 76 };
 77 
 78 /** 
 79  * displayArray adds colors to colorList and displays them with their corresponding colors.
 80  * @param {array} array - the array of colors
 81  */
 82 Question.prototype.displayArray = (array) => {
 83     let colorList, color;
 84     for (i = 0; i < array.length; i++) {
 85         if (array[i] == "black") {
 86             color = array[i];
 87             colorList = chalk.keyword(color).bgWhite(color) + " ";
 88         }
 89         else {
 90             color = array[i];
 91             colorList = colorList + chalk.keyword(color)(color) + " ";
 92         }
 93     }
 94     console.log('\n'+colorList+'\n');
 95 };
 96 
 97 /**
 98  * validate function checks that value is included in array. If it is display answer to user
 99  * and ask first question again. If not then display error message and ask colorQuestion.
100  * @param {string} value - user input value
101  * @param {array} array - color array 
102  */
103 Question.prototype.validate = (value, array) => {
104     if (array.includes(value) === true) {
105         console.log('You picked color: '+chalk.keyword(value)(value)+', what a beautiful color!\n');
106         question.promptUser();
107     }
108     else {
109         console.log(chalk.red('Enter correct color!\n'));
110         question.colorQuestion();
111     }
112 };
113 
114 /**
115  * displayMessage displays different messages to user. If an error message, display 
116  * will be on red, if other message will be green. Ask first question again.
117  * @param {string} message - message that is displayed to user
118  * @param {string} value - user's input value
119  * @param {string} type - type of message
120  */
121 Question.prototype.displayMessage = (message, value, type) => {
122     if(type == 'error') {
123         console.log(chalk.red(message, value + '\n'));
124     }
125     else {
126         console.log(chalk.green(message, value + '\n'));
127     }
128     question.promptUser();
129 };
130 
131 module.exports = Question;
