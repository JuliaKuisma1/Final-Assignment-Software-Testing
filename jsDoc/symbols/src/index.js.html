  6 /** chalk library @tutorial https://github.com/chalk/chalk */
  7 const chalk = require('chalk');
  8 /** figlet library @tutorial https://github.com/patorjk/figlet.js#readme */
  9 const figlet = require('figlet');
 10 /** clear library @tutorial https://github.com/bahamas10/node-clear#readme*/
 11 const clear = require('clear');
 12 /**
 13  * @file Console Application, Question module, that runs question logic. 
 14  * @author Julia Kuisma <julia.kuisma@student.lamk.fi>
 15  */
 16 const Question = require('./question')
 17 
 18 /** 
 19  * startUp function starts all functionalities of application. It calls clear and console log to 
 20  * output title and user options. Then it calls question module and promptUser.
 21  */
 22 startUp = () => {
 23     clear();
 24     console.log(
 25         chalk.cyanBright.bgBlue(figlet.textSync('SOFTWARE TESTING', {horizontalLayout: 'full'}))+
 26         chalk.black('\n')+
 27         chalk.magentaBright(figlet.textSync('OPTIONS:', {horizontalLayout: 'full'}))+
 28         chalk.white('\n a - pick a '+chalk.red('c')+chalk.yellow('o')+chalk.green('l')+chalk.cyan('o')
 29                 +chalk.blue('r')+'\n b - favorite sport\n c - close software\n')
 30     );
 31     question = new Question();
 32     question.promptUser();
 33 };
 34 
 35 startUp();
